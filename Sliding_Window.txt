class Solution{   
public:
    int maximumSumSubarray(int K, vector<int> &Arr , int N){
        int sum=0, maxm=0;
        for(int i=0, j=0; j<N; j++){
            if(j-i+1 < K){
                sum=sum+Arr[j];
                continue;
            }
            else if(j-i+1==K){
                sum=sum+Arr[j];
                maxm=max(sum, maxm);
                sum=sum-Arr[i];
                i++;
            }
        }
        return maxm;
    }
};

vector<long long> printFirstNegativeInteger(long long int A[], long long int N, long long int K) {
    vector<long long> result;
    vector<long long> first;
    for(long long int i=0,j=0; j<N; j++){
        if(j-i+1<K){
            if(A[j]<0 ){
                first.push_back(j);
            }
        }
        else if(j-i+1==K){
            if(A[j]<0){
                first.push_back(j);
            }
            if(first.empty()){
                result.push_back(0);
            }
            else{
                result.push_back(A[first.front()]);
                if(i==first.front()){
                first.erase(first.begin());
            }
            }
            
            i++;
        }
    }
    return result;
 }
